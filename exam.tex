\newcommand{\points}[1]{
	\addtocounter{points}{#1}
	\ifnum #1=1
		(#1~Point)
	\else
		(#1~Points)
	\fi
}

	\refstepcounter{task}
  	\begin{flushleft}
		{\LARGE\bf Task \arabic{task}}
	\end{flushleft}
	{\Large \textbf{#1}}
	\leavevmode\hfill (\nextpoints{}~Points)\\
}

\newcounter{points}
\newcounter{nextpoints}
\newcommand{\setuppoints}{%
  \stepcounter{nextpoints}%
  \setcounter{points}{0}%
%  \preto\item{}%
}
\makeatletter
\newcommand{\computepoints}{%
  \edef\@currentlabel{\number\c@points}%
  \label{points@\number\numexpr\value{nextpoints}\relax}%
}
\newcommand{\nextpoints}{%
  \getrefnumber{points@\number\c@nextpoints}%
}
\makeatother
\AtBeginEnvironment{task}{\setuppoints}
\AtEndEnvironment{task}{\computepoints}
